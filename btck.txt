create table users
(
email_user varchar(50) not null primary key,
name_user varchar(50) not null,
password_user varchar(50) not null,
kind int not null
);

create table habits
(
habit_id INT primary key NOT NULL,
name_habit varchar(255) not null ,
email_user varchar(50) not null,
habit_type varchar(50) not null,
start_date date not null,
end_date date not null,
repeat_interval int not null,
foreign key (email_user) references users(email_user)

);


create table goals
(
goal_id int not null primary key,
habit_id int not null,
goal_type varchar(50) not null,
frequency varchar(50) not null,
goal_target int not null,
foreign key (habit_id) references habits(habit_id)
);
CREATE TABLE reminders
(
reminder_id INT NOT NULL PRIMARY KEY,
label_habit varchar(255) not null,
habit_id INT NOT NULL,
reminder_time TIME,
FOREIGN KEY (habit_id) REFERENCES habits(habit_id)
);
CREATE TABLE completed_habits
(
completed_id INT NOT NULL PRIMARY KEY,
habit_id INT NOT NULL,
completion_date DATE,
progress int not null,
complete_status varchar(50) not null,
FOREIGN KEY (habit_id) REFERENCES habits(habit_id)
);
